--- contoh koneksi nginx docker dengan container lain + ssl certificate

docker run --name nginx1 --link app -p 80:80 -p 443:443 -v ~/application/nginx1/nginx.conf:/etc/nginx/nginx.conf -v ~/application/nginx1/ssl/certs/cert.pem:/etc/nginx/ssl/certs/cert.pem -v ~/application/nginx1/ssl/private/cert.key:/etc/nginx/ssl/private/cert.key nginx:stable-alpine
